📡 Extract POS Sales — Incremental Load Data Flow

-- Connection: POS-SERVER → POSTransactions (OLEDB/SQLNCLI11.1)
-- Target:     DW-SQL-SERVER → BrightfieldDW staging.POSSales

── Step 1: Get Last Extract Date ──────────────────────────
SELECT MAX(ExtractDate) FROM audit.ExtractLog
WHERE SourceSystem = 'POS'
→ stores result in @LastExtractDate variable

── Step 2: Data Flow Task ─────────────────────────────────

  ┌─────────────────────────────────────────────────┐
  │  OLE DB Source: POS Sales Source                │
  │                                                   │
  │  SELECT TransactionID, StoreID, ProductID,       │
  │         CustomerID, TransactionDate,              │
  │         Quantity, UnitPrice, Discount, TotalAmount │
  │  FROM dbo.SalesTransactions                      │
  │  WHERE TransactionDate > @LastExtractDate         │
  └───────────────────────┬─────────────────────────┘
                          │
                          ▼
  ┌─────────────────────────────────────────────────┐
  │  Derived Column: Add Audit Columns              │
  │                                                   │
  │  ExtractDate  = GETDATE()                       │
  │  SourceSystem = "POS"                            │
  └───────────────────────┬─────────────────────────┘
                          │
                          ▼
  ┌─────────────────────────────────────────────────┐
  │  OLE DB Destination: staging.POSSales            │
  │  FastLoad: TABLOCK, CHECK_CONSTRAINTS            │
  └─────────────────────────────────────────────────┘

── Step 3: Log Extract Completion ─────────────────────────
INSERT INTO audit.ExtractLog (SourceSystem, ExtractDate, RowCount)
VALUES ('POS', GETDATE(), @RowCount)
