🔄 Incremental Load — High Water Mark Pattern

-- Watermark tracking table: remembers "where we left off"

CREATE TABLE audit.WatermarkTable (
    SourceTable    VARCHAR(255) PRIMARY KEY,
    WatermarkColumn VARCHAR(100),
    WatermarkValue  SQL_VARIANT,
    LastUpdateDate  DATETIME DEFAULT GETDATE()
);

-- Pre-populated watermarks
INSERT INTO audit.WatermarkTable VALUES
  ('POSSales',        'TransactionDate',  '2024-01-01'),
  ('InventoryLevels', 'SnapshotDate',     '2024-01-01'),
  ('CRMCustomers',    'LastModifiedDate', '2024-01-01');

-- Get watermark before extract
CREATE PROCEDURE audit.GetWatermark
    @SourceTable VARCHAR(255)
AS BEGIN
    SELECT WatermarkValue, LastUpdateDate
    FROM audit.WatermarkTable
    WHERE SourceTable = @SourceTable;
END

-- Update watermark after extract
CREATE PROCEDURE audit.UpdateWatermark
    @SourceTable  VARCHAR(255),
    @NewWatermark SQL_VARIANT
AS BEGIN
    UPDATE audit.WatermarkTable
    SET WatermarkValue = @NewWatermark,
        LastUpdateDate = GETDATE()
    WHERE SourceTable = @SourceTable;
END

-- POS extract query using watermark
SELECT ISNULL(MAX(TransactionDate), '1900-01-01') AS LastExtractDate
FROM staging.POSSales
WHERE SourceSystem = 'POS';
