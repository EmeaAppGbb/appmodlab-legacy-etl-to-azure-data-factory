📥 Load FactSales — Drop/Rebuild Index Strategy

① Drop Indexes  — faster bulk insert without index maintenance
   EXEC dw.DropFactSalesIndexes
   -- Drops: IX_FactSales_DateKey, _StoreKey, _ProductKey, _CustomerKey
        │
        ▼
② Prepare Target  — incremental: no-op / full reload: TRUNCATE
        │
        ▼
③ Merge Sales Data
   EXEC dw.MergeFactSales

   INSERT INTO dw.FactSales (...)
   SELECT
       d.DateKey, st.StoreKey, p.ProductKey, c.CustomerKey,
       s.TransactionID, s.Quantity, s.UnitPrice,
       s.Discount,
       (s.Quantity * s.UnitPrice) * (s.Discount / 100.0) AS DiscountAmount,
       s.Quantity * s.UnitPrice                          AS Revenue,
       p.UnitCost * s.Quantity                           AS Cost,
       (Revenue) - (Cost)                                AS GrossProfit,
       s.TotalAmount                                     AS NetRevenue
   FROM staging.POSSales s
   JOIN dw.DimDate     d  ON CAST(s.TransactionDate AS DATE) = d.Date
   JOIN dw.DimStore    st ON s.StoreID = st.StoreID
   JOIN dw.DimProduct  p  ON s.ProductID = p.ProductID AND p.IsCurrent = 1
   LEFT JOIN dw.DimCustomer c ON s.CustomerID = c.CustomerID AND c.IsCurrent = 1
   WHERE NOT EXISTS (SELECT 1 FROM dw.FactSales f
                     WHERE f.TransactionID = s.TransactionID)
     AND s.Quantity > 0 AND s.UnitPrice > 0;
        │
        ▼
④ Rebuild Indexes
   EXEC dw.RebuildFactSalesIndexes
   -- Recreates all 5 indexes on FactSales
        │
        ▼
⑤ Update Statistics
   UPDATE STATISTICS dw.FactSales
