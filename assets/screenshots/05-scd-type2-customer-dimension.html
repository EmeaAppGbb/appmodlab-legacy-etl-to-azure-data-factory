🔄 SCD Type 2 — Slowly Changing Customer Dimension

CREATE PROCEDURE dw.ProcessCustomerDimensionSCD2
AS BEGIN
    SET NOCOUNT ON;

    -- Step 1: Expire changed records
    UPDATE t
    SET
        t.EndDate        = GETDATE(),
        t.IsCurrent      = 0,
        t.RecordModified = GETDATE()
    FROM dw.DimCustomer t
    INNER JOIN staging.CRMCustomers s
        ON t.CustomerID = s.CustomerID
    WHERE t.IsCurrent = 1
      AND (
          t.FullName    <> s.FullName OR
          t.Email       <> s.Email OR
          t.LoyaltyTier <> s.LoyaltyTier OR
          t.Segment     <> s.Segment OR
          t.Address     <> s.Address
      );

    -- Step 2: Insert new versions of changed records
    INSERT INTO dw.DimCustomer (
        CustomerID, FullName, Email, Phone, Address,
        City, State, Segment, LoyaltyTier, LoyaltyPoints,
        StartDate, EndDate, IsCurrent
    )
    SELECT
        s.CustomerID, s.FullName, s.Email, s.Phone, s.Address,
        s.City, s.State, s.Segment, s.LoyaltyTier, s.LoyaltyPoints,
        GETDATE(), NULL, 1
    FROM staging.CRMCustomers s
    INNER JOIN dw.DimCustomer t ON s.CustomerID = t.CustomerID
    WHERE t.IsCurrent = 0
      AND t.EndDate = (SELECT MAX(EndDate) FROM dw.DimCustomer
                       WHERE CustomerID = s.CustomerID);

    -- Step 3: Insert completely new customers
    INSERT INTO dw.DimCustomer (...)
    SELECT s.* FROM staging.CRMCustomers s
    WHERE NOT EXISTS (
        SELECT 1 FROM dw.DimCustomer t
        WHERE t.CustomerID = s.CustomerID
    );
END
